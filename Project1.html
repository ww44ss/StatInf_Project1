##create data for analysis

        ##Set seed to make results reproducible
        set.seed(54321)

        ##makes sample sizes variable
        rowsofsamples <- 40
        columnsofdistributions <- 1000

        ##create data frame
        rdata <- replicate(columnsofdistributions, rexp(rowsofsamples, .2)) 
       
        ##look at structure of data
        str(rdata)
##  num [1:40, 1:1000] 4.08 3.95 7.1 10.3 5.3 ...
##Take means of columns and plot histrogram with mean and theoretical mean 
        
        ##Compute Column means
        expmeans <- colMeans(rdata)

        ##Generate Historgram
        hist(expmeans, density=20, angle=45) #, xlim=c(4.6,5.4), ylim=c(0,11))

        ##Calculate "Mean of Means"
        meanofdistmeans<-mean(expmeans)
        
        ##Draw lines at mean and theoretical mean = 5
        abline(v=meanofdistmeans, lwd=3, col="red")
        abline(v=5, lwd=3, col="blue")

        ##Label graph
        text(meanofdistmeans,10.5, "Data Mean", col="red", adj=c(1.1,0))
        text(5,10.5, "Theoretical Mean", col="blue", adj=c(-.10,0))

varofdistmeans = var(expmeans)
